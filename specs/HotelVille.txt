service : HotelVille
observators :
	const largeur : [HotelVille] → int
	const hauteur : [HotelVille] → int
	const posx : [HotelVille] -> int
	const posy : [HotelVille] -> int
	orRestant : [HotelVille] → int
	
Constructors :
	init : int × int x int × int  → [HotelVille]
		pre init(x,y,largeur,hauteur,or) require 
			largeur%2=1 ∧ 
			hauteur%2=1 ∧
			0 <= posx <= largeur ∧
			0 <= posy <= hauteur ∧
		    or > 0
Operators :
	depot : [HotelVille] × int → [HotelVille]
		pre depot(H,s) require s>0

		
Observations :
	[init]
	posx(init(x,y,l,h,or))=x
	posy(init(x,y,l,h,or))=y
	largeur(init(x,y,l,h,or))=l
	hauteur(init(x,y,l,h,or))=h
	orRestant(init(x,y,l,h,or))=or

	
	[depot]
	orRestant(depot(H,s))=orRestant(H)@pre+s
